        saldo = 0;
        Status = False;

        return tipo;

    public func getDono() {
        return dono;
    }

    public func setSaldo(Float) {
        saldo = s;
    }

    public func getSaldo() {
        return saldo;
    }

    public func setStatus(booleano) {
        Status = st;
    }

    public func getStatus() {
        return Status;

    // METODOS ESPECIFICOS: //

    public func AbrirConta(t) {
        setTipo(t);
        setStatus(True);

        if (t == ContaCorrente) {
            saldo = 50;
        }

        if (t == ContaPoupança) {
            saldo = 150;
        }
    }

    public func FecharConta() {
        if (saldo > 0) {
            print("ERRO: ESSA CONTA AINDA TEM DINHEIRO!")
        }

        if (saldo < 0) {
            print("ERRO: VOCÊ ESTÁ DEVENDO DINHEIRO!")
        }

        else {
            print("FECHAMENTO EFETUADO COM SUCESSO!")
            setStatus(False);
        }
    }

    public func Sacar(v) {
        if (Status == True and saldo > 0) {
            saldo = saldo - v;
        }

        if (saldo < 0) {
            print("ERRO: SALDO INSUFICIENTE!")
        }

        if (Status == False) {
            print("ERRO: IMPOSSIVEL SACAR!")
        }
    }

    public func Depositar(v) {
        if (Status == True) {
            saldo = saldo + v;
        }

        else {
            print("ERRO: IMPOSSIVEL DEPOSITAR")
        }
    }

    public func PagarMensal(v) {
        var v = Float;
        if (t == ContaCorrente) {
            v = 12;
        }

        if (t == ContaPoupança) {
            v = 20;
        }

        if (Status == True) {
            if (saldo > v) {
                saldo = saldo - v;
            }
            else {
                print("ERRO: SALDO INSUFICIENTE!")
            }
        }
        else {
            print("ERRO: IMPOSSIVEL PAGAR!")
        }
    }
}
